public class Condicional {
    public static void main(String[] args) {
        int x = 10;
        if (x > 5) {
            System.out.println("x es mayor que 5");
        } else {
            System.out.println("x es menor o igual que 5");
        }

        for (int i = 0; i < x; i++) {
            System.out.println("i = " + i);
        }
    }
}
